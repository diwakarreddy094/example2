CREATE OR REPLACE PROCEDURE ADDCUSTOMER(CUSTOMERNUMBER IN CUSTOMER.CUSTOMERNUMBER%TYPE,
                             CUSTOMERNAME IN CUSTOMER.CUSTOMERNAME%TYPE,
                             EMAIL IN CUSTOMER.EMAIL%TYPE,
                             HANDPHONE IN CUSTOMER.HANDPHONE%TYPE)
IS
BEGIN
 INSERT INTO CUSTOMER VALUES(CUSTOMERNUMBER,CUSTOMERNAME,EMAIL,HANDPHONE);
END

END;


CREATE OR REPLACE PROCEDURE CUSTOMERQUERY(CUSTOMERNUMBER1 IN CUSTOMER.CUSTOMERNUMBER%TYPE,
                             CUSTOMERNAME1 OUT CUSTOMER.CUSTOMERNAME%TYPE,
                             EMAIL1 OUT CUSTOMER.EMAIL%TYPE,
                             HANDPHONE1 OUT CUSTOMER.HANDPHONE%TYPE)
IS
BEGIN
 SELECT CUSTOMERNAME,EMAIL,HANDPHONE  INTO CUSTOMERNAME1,EMAIL1,HANDPHONE1 FROM CUSTOMER WHERE CUSTOMERNUMBER=CUSTOMERNUMBER1;
EXCEPTION
 WHEN NO_DATA_FOUND THEN RAISE_APPLICATION_ERROR(-20001,'NO RECORDS EXIST');
END CUSTOMERQUERY;


DECLARE
CUSTOMERNUMBER2 CUSTOMER.CUSTOMERNAME%TYPE;
EMAIL2 CUSTOMER.EMAIL%TYPE;
HANDPHONE2 CUSTOMER.HANDPHONE%TYPE;
BEGIN
 CUSTOMERQUERY(101,CUSTOMERNUMBER2,EMAIL2,HANDPHONE2);
   DBMS_OUTPUT.PUT_LINE(EMAIL2);
   DBMS_OUTPUT.PUT_LINE(HANDPHONE2);
   DBMS_OUTPUT.PUT_LINE(CUSTOMERNUMBER2);
END;
